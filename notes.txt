wget https://www.libssh.org/files/0.9/libssh-0.9.6.tar.xz
tar xf libssh-0.9.6.tar.xz
cd libssh-0.9.6
mkdir build
cd build
cmake .. -DCMAKE_BUILD_TYPE=Release -DBUILD_STATIC_LIB=ON -DWITH_GSSAPI=OFF
make
sudo make install
